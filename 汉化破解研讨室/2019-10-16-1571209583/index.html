    Quartett!的文本提取程序

      博客分类：
    • Reverse Engineering
    CC++C#生物脚本
    诶，之前写了这个程序却没有发出来，结果自己差点找不到了。幸好在笔记本的某个不起眼的角落里又见到了它啊 T T

    这个程序用于从*.tkn脚本文件中提取“文本”（text=之后的项）。对应的ScriptInserter那时没写……

    提取出来的文本像这样：
    引用
    135
    <FONT size=22 face='G'><TYPE interval=100% speed=700 style='Wave'>心配だな～
    167
    <TYPE interval=60>お兄ちゃん<BR>...<TYPE interval=55>ほんとに大丈夫なの？
    170
    <FONT size=16><TYPE interval=60>今日のクリスマス・ミサには...<BR>毎年..たくさんの人が来るんだよ


    一行原始行号，一行文本。行号是用于校验用的。

    28. using System;  
    29. using System.Collections.Generic;  
    30. using System.IO;  
    31. using System.Text;  
    32.   
    33. namespace FFDSystemAnalysis  
    34. {  
    35.     enum TokenType  
    36.     {  
    37.         Decimal = 0x080,  
    38.         Identifier = 0x081,  
    39.         Hexadecimal = 0x082,  
    40.         String = 0x083,  
    41.         Operator = 0x085  
    42.     }  
    43.       
    44.     enum OutputState  
    45.     {  
    46.         Searching,  
    47.         FoundText,  
    48.         FoundEqual  
    49.     }  
    50.   
    51.     sealed class ScriptExtractor  
    52.     {  
    53.         private static readonly byte[ ] SIGNATURE = {  
    54.             ( byte )0x54, ( byte )0x4F, ( byte )0x4B, ( byte )0x45,  
    55.             ( byte )0x4E, ( byte )0x53, ( byte )0x45, ( byte )0x54,  
    56.             ( byte )0x64, ( byte )0x0,  ( byte )0x0,  ( byte )0x0  
    57.         };  
    58.   
    59.         static void Main( string[ ] args ) {  
    60.             if ( !args[ 0 ].EndsWith( ".tkn" ) ) return;  
    61.             if ( !File.Exists( args[ 0 ] ) ) return;  
    62.   
    63.             string infile = args[ 0 ];  
    64.             string outfile = Path.GetFileNameWithoutExtension( infile ) + ".txt";  
    65.   
    66.             Encoding utf16le = new UnicodeEncoding( false, true );  
    67.             Encoding jis = Encoding.GetEncoding( 932 );  
    68.   
    69.             using ( BinaryReader reader = new BinaryReader( File.OpenRead( infile ), jis ) ) {  
    70.                 using ( BinaryWriter writer = new BinaryWriter( File.Create( outfile ), utf16le ) ) {  
    71.                     byte[ ] sig = reader.ReadBytes( SIGNATURE.Length );  
    72.                     if ( !Equals( sig, SIGNATURE ) ) {  
    73.                         Console.WriteLine( "Wrong signature" );  
    74.                         return;  
    75.                     }  
    76.   
    77.                     // write UTF-16LE BOM  
    78.                     writer.Write( ( ushort ) 0xFEFF );  
    79.   
    80.                     // process each token  
    81.                     int lineNum = 1;  
    82.                     TokenType tokenType = TokenType.Operator;  
    83.                     OutputState state = OutputState.Searching;  
    84.                     int tokenCount = reader.ReadInt32( );  
    85.                     for ( int tokenNum = 0; tokenNum < tokenCount; ++tokenNum ) {  
    86.                         // get line numbers  
    87.                         lineNum = reader.ReadInt32( );  
    88.   
    89.                         // get token tokenType  
    90.                         tokenType = ( TokenType ) ( reader.ReadByte( ) & 0x0FF );  
    91.   
    92.                         // get token value  
    93.                         string tokenString = ReadCString( reader );  
    94.   
    95.                         switch ( tokenType ) {  
    96.                           
    97.                         case TokenType.Identifier:  
    98.                             if ( tokenString.Equals("text") ) {  
    99.                                 state = OutputState.FoundText;  
    100.                             } else {  
    101.                                 state = OutputState.Searching;  
    102.                             }  
    103.                             break;  
    104.   
    105.                         case TokenType.String:  
    106.                             if (OutputState.FoundEqual == state)  
    107.                             {  
    108.                                 writer.Write( utf16le.GetBytes( lineNum.ToString( ) ) );  
    109.                                 writer.Write( utf16le.GetBytes( Environment.NewLine ) );  
    110.                                 writer.Write( utf16le.GetBytes( tokenString ) );  
    111.                                 writer.Write( utf16le.GetBytes( Environment.NewLine ) );  
    112.                                 state = OutputState.Searching;  
    113.                             }  
    114.                             break;  
    115.   
    116.                         case TokenType.Operator:  
    117.                             if ((OutputState.FoundText == state)  
    118.                                 && tokenString.Equals("=")) {  
    119.                                 state = OutputState.FoundEqual;  
    120.                             } else {  
    121.                                 state = OutputState.Searching;  
    122.                             }  
    123.                             break;  
    124.                           
    125.                         case TokenType.Decimal:  
    126.                         case TokenType.Hexadecimal:  
    127.                             // do nothing  
    128.                             break;  
    129.                               
    130.                         default:  
    131.                             Console.WriteLine( "Unexpected token type {0} at 0x{1}.",  
    132.                                 tokenType.ToString( "X" ),  
    133.                                 reader.BaseStream.Position.ToString( "X" ) );  
    134.                             return;  
    135.                         } // switch tokenType  
    136.                     } // for  
    137.                 }  
    138.             }  
    139.         }  
    140.   
    141.         static bool Equals( byte[ ] a, byte[ ] b ) {  
    142.             int len = a.Length;  
    143.             if ( len != b.Length ) return false;  
    144.             for ( int i = 0; i < len; i++ ) {  
    145.                 if ( a[ i ] != b[ i ] ) return false;  
    146.             }  
    147.             return true;  
    148.         }  
    149.   
    150.         static string ReadCString( BinaryReader reader ) {  
    151.             StringBuilder builder = new StringBuilder( );  
    152.             char c = '\0';  
    153.   
    154.             while ( ( c = reader.ReadChar( ) ) != '\0' ) {  
    155.                 builder.Append( c );  
    156.             }  
    157.   
    158.             return builder.ToString( );  
    159.         }  
    160.     }  
    161. }
