    这东西总算也结束了……我基本上可以安心离开汉化圈子了。

    嘛，这脚本跟前面的BMW、夜明的脚本基本上是一样的，所以也没什么太多可解释的。
    而且继续偷懒……懒得做目录操作所以那graph目录我做成了一个自解压exe文件，复制到安装目录后自动展开。

    NSIS脚本代码：
    Java代码  
    1. SetCompressor lzma  
    2.   
    3. ; HM NIS Edit Wizard helper defines  
    4. !define PRODUCT_NAME "七色★星露 汉化补丁"  
    5. !define PRODUCT_VERSION "1.0"  
    6. !define PRODUCT_PUBLISHER "奥克汉化组"  
    7. !define PRODUCT_DIR_REGKEY "Software\Microsoft\Windows\CurrentVersion\App Paths\yoake_chs"  
    8. !define PRODUCT_UNINST_KEY "Software\Microsoft\Windows\CurrentVersion\Uninstall\${PRODUCT_NAME}"  
    9. !define PRODUCT_UNINST_ROOT_KEY "HKLM"  
    10. !define PRODUCT_STARTMENU_REGVAL "NSIS:StartMenuDir"  
    11.   
    12. ; MUI 1.67 compatible ------  
    13. !include "MUI.nsh"  
    14. !define MUI_WELCOMEFINISHPAGE_BITMAP ".\image\side.bmp"  
    15. !include "FileFunc.nsh"  
    16. !include "nsisbass.nsh"  
    17. !include "nsisbassmacros.nsh"  
    18. !insertmacro DirState  
    19.   
    20. ; MUI Settings  
    21. !define MUI_ABORTWARNING  
    22. !define MUI_ICON ".\icon\nana.ico"  
    23. !define MUI_UNICON ".\icon\nana.ico"  
    24.   
    25. ; Language Selection Dialog Settings  
    26. !define MUI_LANGDLL_REGISTRY_ROOT "${PRODUCT_UNINST_ROOT_KEY}"  
    27. !define MUI_LANGDLL_REGISTRY_KEY "${PRODUCT_UNINST_KEY}"  
    28. !define MUI_LANGDLL_REGISTRY_VALUENAME "NSIS:Language"  
    29.   
    30. ; Welcome page  
    31. !insertmacro MUI_PAGE_WELCOME  
    32. ; License page  
    33. !define MUI_LICENSEPAGE_CHECKBOX  
    34. !insertmacro MUI_PAGE_LICENSE ".\txt\license.txt"  
    35. ; Directory page  
    36. !insertmacro MUI_PAGE_DIRECTORY  
    37. ; Start menu page  
    38. var /GLOBAL ICONS_GROUP  
    39. !define MUI_STARTMENUPAGE_DEFAULTFOLDER "奥克汉化组"  
    40. !define MUI_STARTMENUPAGE_REGISTRY_ROOT "${PRODUCT_UNINST_ROOT_KEY}"  
    41. !define MUI_STARTMENUPAGE_REGISTRY_KEY "${PRODUCT_UNINST_KEY}"  
    42. !define MUI_STARTMENUPAGE_REGISTRY_VALUENAME "${PRODUCT_STARTMENU_REGVAL}"  
    43. !insertmacro MUI_PAGE_STARTMENU Application $ICONS_GROUP  
    44. ; Instfiles page  
    45. !insertmacro MUI_PAGE_INSTFILES  
    46. ; Finish page  
    47. !define MUI_FINISHPAGE_RUN "$INSTDIR\nana_chs.exe"  
    48. !insertmacro MUI_PAGE_FINISH  
    49.   
    50. ; Uninstaller pages  
    51. !insertmacro MUI_UNPAGE_INSTFILES  
    52.   
    53. ; Language files  
    54. !insertmacro MUI_LANGUAGE "SimpChinese"  
    55. VIProductVersion "1.0.0.0"  
    56. VIAddVersionKey /LANG=2052 "FileDescription" "${PRODUCT_NAME}"  
    57. VIAddVersionKey /LANG=2052 "ProductName" "${PRODUCT_NAME}"  
    58. VIAddVersionKey /LANG=2052 "LegalCopyright" ""  
    59. VIAddVersionKey /LANG=2052 "CompanyName" "${PRODUCT_PUBLISHER}"  
    60. VIAddVersionKey /LANG=2052 "FileVersion" "${PRODUCT_VERSION}"  
    61.   
    62. ReserveFile ".\image\splash.bmp"  
    63. ReserveFile "${NSISDIR}\Plugins\system.dll"  
    64. ReserveFile "${NSISDIR}\Contrib\nsisbass\bass.dll"  
    65. ReserveFile ".\music\bgm.mp3"  
    66.   
    67. ; MUI end ------  
    68.   
    69. Name "${PRODUCT_NAME} ${PRODUCT_VERSION}"  
    70. OutFile "nana_patch_chs.exe"  
    71. InstallDir "$EXEDIR"  
    72. InstallDirRegKey HKLM "${PRODUCT_DIR_REGKEY}" ""  
    73. ShowInstDetails show  
    74. ShowUnInstDetails show  
    75.   
    76. Function .onInit  
    77.   InitPluginsDir  
    78.   File "/oname=$PLUGINSDIR\Splash_Splash.bmp" ".\image\splash.bmp"  
    79.   ; 使用闪屏插件显示闪屏  
    80.   advsplash::show 1500 600 400 -1 "$PLUGINSDIR\Splash_Splash"  
    81.   Pop $0 ; $0 返回 '1' 表示用户提前关闭闪屏, 返回 '0' 表示闪屏正常结束, 返回 '-1' 表示闪屏显示出错  
    82.   ;!insertmacro MUI_LANGDLL_DISPLAY  
    83.   StrCpy $INSTDIR "C:\ユニゾンシフト\ななついろ★ドロップス"  
    84.   ; Insert the NSISBASS_INIT macro  
    85.   !insertmacro NSISBASS_INIT  
    86.   ; We can extract a song file as well (so you got something to play).  
    87.   File "/oname=$PLUGINSDIR\tempsong.mp3" ".\music\BGM.mp3"  
    88.   !insertmacro NSISBASS_PLAY "$PLUGINSDIR\tempsong.mp3"  
    89. FunctionEnd  
    90.   
    91. Function .onVerifyInstDir  
    92.   IfFileExists "$INSTDIR\nana.exe" PathGood  
    93.     Abort ; if $INSTDIR is not a NanaDoro directory, don't let us install there  
    94.   PathGood:  
    95. FunctionEnd  
    96.   
    97.   
    98. Section "MainSection" SEC01  
    99. ;--------------------------copy----------------------------  
    100.   SetOutPath "$INSTDIR"  
    101.   DetailPrint "正在复制文件..."  
    102.   SetOverwrite ifnewer  
    103.   File ".\files\nana_chs.exe"  
    104.   File ".\files\scenario.dat"  
    105.   File ".\files\textdata.dat"  
    106.   File ".\files\GAME.ini"  
    107.   File ".\files\graph.exe"  
    108.   ExecWait '"$INSTDIR\graph.exe"'  
    109.   Delete "$INSTDIR\graph.exe"  
    110. SectionEnd  
    111.   
    112. Section "Add Start Menu Shortcuts" SEC02  
    113.   ; shortcuts  
    114.   !insertmacro MUI_STARTMENU_WRITE_BEGIN Application  
    115.   SetOutPath "$INSTDIR"  
    116.   CreateDirectory "$SMPROGRAMS\$ICONS_GROUP"  
    117.   CreateShortCut "$SMPROGRAMS\$ICONS_GROUP\七色★星露 汉化版.lnk" "$INSTDIR\nana_chs.exe"  
    118.   CreateShortCut "$DESKTOP\七色★星露 汉化版.lnk" "$INSTDIR\nana_chs.exe"  
    119.   !insertmacro MUI_STARTMENU_WRITE_END  
    120. SectionEnd  
    121.   
    122. Section -AdditionalIcons  
    123.   !insertmacro MUI_STARTMENU_WRITE_BEGIN Application  
    124.   CreateShortCut "$SMPROGRAMS\$ICONS_GROUP\卸载简体中文补丁.lnk" "$INSTDIR\uninst.exe"  
    125.   !insertmacro MUI_STARTMENU_WRITE_END  
    126. SectionEnd  
    127.   
    128. Section -Post  
    129.   WriteUninstaller "$INSTDIR\uninst.exe"  
    130.   WriteRegStr HKLM "${PRODUCT_DIR_REGKEY}" "" "$INSTDIR\nana_chs.exe"  
    131.   WriteRegStr ${PRODUCT_UNINST_ROOT_KEY} "${PRODUCT_UNINST_KEY}" "DisplayName" "$(^Name)"  
    132.   WriteRegStr ${PRODUCT_UNINST_ROOT_KEY} "${PRODUCT_UNINST_KEY}" "UninstallString" "$INSTDIR\uninst.exe"  
    133.   WriteRegStr ${PRODUCT_UNINST_ROOT_KEY} "${PRODUCT_UNINST_KEY}" "DisplayIcon" "$INSTDIR\nana_chs.exe"  
    134.   WriteRegStr ${PRODUCT_UNINST_ROOT_KEY} "${PRODUCT_UNINST_KEY}" "DisplayVersion" "${PRODUCT_VERSION}"  
    135.   WriteRegStr ${PRODUCT_UNINST_ROOT_KEY} "${PRODUCT_UNINST_KEY}" "Publisher" "${PRODUCT_PUBLISHER}"  
    136. SectionEnd  
    137.   
    138.   
    139. Function un.onUninstSuccess  
    140.   HideWindow  
    141.   MessageBox MB_ICONINFORMATION|MB_OK "「七色★星露 汉化补丁」已成功地从你的计算机删除。"  
    142. FunctionEnd  
    143.   
    144. Function un.onInit  
    145. !insertmacro MUI_UNGETLANGUAGE  
    146.   MessageBox MB_ICONQUESTION|MB_YESNO|MB_DEFBUTTON2 "你确实要完全删除「七色★星露 汉化补丁」，其及所有組件？" IDYES +2  
    147.   Abort  
    148. FunctionEnd  
    149.   
    150. Section Uninstall  
    151.   !insertmacro MUI_STARTMENU_GETFOLDER "Application" $ICONS_GROUP  
    152. ;----------------------delete_files------------------------  
    153.   Delete "$INSTDIR\uninst.exe"  
    154.   Delete "$INSTDIR\GAME.ini"  
    155.   Delete "$INSTDIR\textdata.dat"  
    156.   Delete "$INSTDIR\scenario.dat"  
    157.   Delete "$INSTDIR\nana_chs.exe"  
    158.   RMDir /r "$INSTDIR\graph"  
    159. ;--------------------delete_shortcuts----------------------  
    160.   Delete "$SMPROGRAMS\$ICONS_GROUP\七色★星露 汉化版.lnk"  
    161.   Delete "$SMPROGRAMS\$ICONS_GROUP\卸载简体中文补丁.lnk"  
    162.   Delete "$DESKTOP\七色★星露 汉化版.lnk"  
    163.   RMDir "$SMPROGRAMS\$ICONS_GROUP"  
    164. ;-----------------delete_uninst_reg_keys-------------------  
    165.   DeleteRegKey ${PRODUCT_UNINST_ROOT_KEY} "${PRODUCT_UNINST_KEY}"  
    166.   DeleteRegKey HKLM "${PRODUCT_DIR_REGKEY}"  
    167. ;--------------------close_after_unist---------------------  
    168.   SetAutoClose true  
    169. SectionEnd
